class laptop:
    storage_type = 'ssd'

    def __init__(self, RAM, storage):
        self.RAM = RAM
        self.srorage = storage

    def get_info(self):
        print(f"laptop has {self.RAM} RAM and {self.srorage} {self.storage_type}")
                                                            ----------------------
l1 = laptop('16gb', '512gb')                                                   |
l2 = laptop('8gb', '256gb')                                                    |
l1.get_info()        # output - laptop has 16gb RAM and 512gb ssd              |
